\section{Introduction}




\begin{frame}{Motivation}
\begin{wideitemize}


\item A central objective in the economics of education is to estimate the returns to different educational programs. Within this broader agenda, a large literature has focused on school value-added. \parencite{angrist_stand_2016, dobbie_medium-term_2015, angrist_leveraging_2017, angrist_credible_2024}. 

\item Much of this work starts by studying effects on academic achievement, though recent research has broadened the notion of value-added to include other outcomes. \parencite[]{beuermann_what_2023, rose_effects_2022}. 

\item By contrast, the effect of schools on major choice, an especially important decision, has received little attention. I propose to study school effects on major choice and to disentangle the extent to which those effects operate through academic achievement.
\end{wideitemize}
\end{frame}


\begin{frame}{Literature Review}
\textbf{Three strands of literature}

\begin{itemize}
    \item \textbf{School effects on higher education outcomes}
    \begin{itemize}
        \item Literature has focused on achievement or the higher educational choices on the `intensive' margin (college enrollment, number of academic semesters) \parencite{chetty_measuring_2014, angrist_stand_2016, dobbie_medium-term_2015, abdulkadiroglu_elite_2014, abdulkadiroglu_parents_2020}.
    \end{itemize}

    \item \textbf{College choice beyond achievement}
    \begin{itemize}
        \item Evidence for margins of college choice  not fully determined by test scores. e.g. lower-income students are less likely to apply to selective institutions than equally high-achieving peers \parencite{dynarski_closing_2021, chetty_diversifying_2025 , hoxby_what_2015}.
    \end{itemize}

    \item \textbf{Major choice and returns}
    \begin{itemize}
        \item Returns to higher education vary substantially across fields \parencite{kirkeboen_field_2016}.
        \item Information about returns can shift students toward higher-return programs even without changing achievement \parencite{hastings_informed_2016}.
    \end{itemize}
\end{itemize}

\vspace{0.4em}
\textbf{This paper:} I connect these three strands by studying how schools affect where students apply, where they are admitted and enroll, and which majors they choose, through both achievement and non-achievement channels.
    
\end{frame}

\begin{frame}{Conceptual Framework}
\begin{itemize}
    \item Assume a constant-effects model for school value added:
    \[
    Y_{ij} = \nu_j + \epsilon_i
    \]
    \item This implies school effects can be written relative to the average school effect:
    \[
    Y_i = \beta_0 + \sum_{j=1}^J \beta_j D_{ij} + \epsilon_i
    \]
    \item The core challenge is selection: students attending a school may differ systematically in unobservables.
\end{itemize}
%\parencite{abdulkadiroglu_research_2017}\parencite{angrist_credible_2024}
\end{frame}




\begin{frame}{Identification from Centralized Assignment}
\begin{itemize}
    \item In centralized assignment, students submit ranked preferences and schools rank students by priority.
    \item Randomness enters through lottery tie-breakers.
    \item Define student type as the combination of preferences and priorities:
    \[
    \theta_i \equiv (\succ_i, \rho_i)
    \]
    \item Under ``equal treatment of equals'' (ETE), students with the same type face the same assignment probabilities.  \parencite{abdulkadiroglu_research_2017}
      \[
    \epsilon_i \perp\!\!\!\perp Z_i \mid \theta_i
    \]
    \item Other (fixed under randomization) variables do not affect the assignment. Key example: $\epsilon_i$. 

\end{itemize}
\end{frame}

\begin{frame}{Assignment Risk as a Control Strategy}
\begin{itemize}
    \item Saturating on exact student type is often infeasible because preference-priority cells are too sparse.
    \item A practical alternative is to use assignment probabilities \(p_i\) as propensity scores. \parencite{abdulkadiroglu_research_2017,rosenbaum_central_1983} 

    \item As a result we can write instead: 
    \[
    \epsilon_i \perp\!\!\!\perp Z_i \mid p_i
    \]
    \item This motivates ``risk-controlled'' designs that condition on school-assignment risk.
\end{itemize}
\end{frame}

\begin{frame}{Example: Common Risk Without Common Type}
\small
\begin{columns}[T]
\begin{column}{0.46\textwidth}
\textbf{Setup}
\begin{itemize}
    \item Applicants: $\{1,2,3,4,5\}$
    \item Schools: $\{a,b,c\}$, one seat each
    \item Priority:
    \begin{itemize}
        \item 5 highest at $c$
        \item 2 highest at $b$
    \end{itemize}
\end{itemize}




\textbf{Preferences}
\[
\begin{array}{ll}
1:& a \succ b\\
2:& a \succ b\\
3:& a\\
4:& c \succ a\\
5:& c
\end{array}
\]
\end{column}

\begin{column}{0.54\textwidth}
\textbf{DA implication}
\begin{itemize}
    \item 5 gets $c$
    \item 1, 2, 3, 4 compete for $a$
    \item none has an advantage at $a$
\end{itemize}

\[
P(Z_{ia}=1)=0.25 \qquad \forall i \in \{1,2,3,4\}
\]

\textbf{Takeaway}
\begin{itemize}
    \item No two applicants share the same full type
    \item Full-type conditioning gives one stratum per applicant
    \item But applicants 1--4 share the same propensity score for $a$
    \item So they can be pooled in one risk stratum
\end{itemize}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{From Offers to Attendance}
\begin{itemize}
    \item Offers \(Z_i\) are exogenous conditional on assignment risk \(p_i\).
    \item To move from offers to actual attendance \(D_i\), the project invokes a conditional-independence argument:
    \[
    \epsilon_i \perp\!\!\!\perp D_i \mid p_i
    \]
    \item Intuition: sorting occurs at the application stage, while post-offer compliance is driven by idiosyncratic shocks. \parencite{dale_estimating_2002} 
    \item This delivers a tractable framework for school-specific reduced-form estimation. \parencite[]{angrist_methods_2023-1}
\end{itemize}

\end{frame}

\begin{frame}{Reduced-Form School Effects}
\begin{itemize}
    \item Example outcomes:
    \begin{itemize}
        \item standardized math achievement,
        \item STEM enrollment in higher education.
    \end{itemize}
    \item Yet also interested in other fields; outcomes such as financial aid application.
    \item Proposed specifications:
    \[
    Y_i^{\text{MATH}} = b_0 + \sum_j b_j D_{ij} + X_i'B + b_{\text{pre}}Y_{\text{pre},i} + g(p_i) + \epsilon_i
    \]
    \[
    Y_i^{\text{STEM}} = c_0 + \sum_j c_j D_{ij} + X_i'C + b_{\text{pre}}Y_{\text{pre},i} + g(p_i) + \eta_i
    \]
    \item Under \textcite{angrist_credible_2024}, these correspond to an RC VAM specification and a risk-controlled OLS.
\end{itemize}
\end{frame}

\begin{frame}{Why Major Choice May Reflect More Than Scores}
\begin{itemize}

    
    \item If math achievement affects STEM enrollment, then school effects on STEM can operate through:
    \begin{enumerate}
        \item a score channel, and
        \item a non-score channel.
    \end{enumerate}
    \item In that case, the reduced-form coefficient on school attendance in the STEM regression mixes both channels.
    \item It could be interesting to focus on schools with estimated zero effects on math scores.
    \item Non-zero STEM effects for those schools could provide evidence of non-score school effects.
\end{itemize}
\end{frame}

\begin{frame}{Potentially: An additional score-only instrument}

\begin{itemize}
    \item In situations like these, the gold-standard for mediation analysis is having access to an instrument that affects only one of the variables (e.g. scores) 
    \item Currently considering using exam forms (randomly assigned, might give a boost of achievement to some students), but concerned about potentially weak first-stage. 

\end{itemize}
    
\end{frame}

 \begin{frame}{Replication of Chilean algorithm}
\begin{itemize}
    \item Any paper on this space needs to pin down the algorithm of assignment.
    \item Managed to replicate above 99\% of original assignments with my 4 sample cohorts. (!) 
    \item Using only student types and a random seed replicated about \(68\%\) of assignments.   
    \item This highlights the importance of correctly modeling the additional institutional features as quotas and guaranteed admission in original school.
    \item Around 30\% of the sample in the centralized system get the same assignment regardless of the lottery number.  
    \item Next steps: explain the remaining mismatch, and try to implement the large market approximation from \textcite{abdulkadiroglu_breaking_2022}, \textcite{abdulkadiroglu_research_2017}.  
\end{itemize}
\end{frame}

\begin{frame}{Discussion: Outside options}
\begin{wideitemize}

\item Unordered treatments: $\beta_s$ I estimate is not relative to a particular option.
\item  \textcite{kirkeboen_field_2016} get around this by using the ``second-choice'' data. Yet they also collapse their effect of spaces to 10 fields (90 effects to estimate).

\item I could try an AKM-intuition \parencite{abowd_high_1999} and identify the effects based on consecutive schools on a given ranking list --akin to movers within 2 firms. But network will likely be divided.

\item \textcite{beuermann_what_2023} offers an interesting test that tackles this issue in their appendix. Test if triangles recover the same effect both ways:  ``i.e. the effect of attending School 1 relative to 5 is equal to the effect of attending School 1 relative to 3 plus the effect of attending School 3 relative to 5''. 


\end{wideitemize}
\end{frame}

\section{Potential next steps}



\begin{frame}{Can this effect be broken down / What are these causal effects correlated to? }

\begin{wideitemize}

\item Unclear what $\beta_s$ is truly capturing given that school is a bundle. I intend to analyze the correlates of these effects in observable characteristics such as teacher quality, staffing and funding.

\item I wonder if it could be worth it to try to complement my data --via surveys or fieldwork-- to unpack these effects? 

\item  Surveying  graduating  students could help unpack the mechanism behind potential school effects (e.g. information on returns; non-tested skills; confidence/role-models/taste formation, etc. )

\item I could also get a measure of traditionally unobserved school attributes E.g. quality of college counseling.

    
\end{wideitemize}
\end{frame}




\begin{frame}{Effect of selection and heterogeneity in school effects}
\begin{itemize}
    \item How constant is  $\beta_s$ ? 

    \item An interesting feature of this setting is that the centralized system of assignment was not always present.
    \item  This might allow me to think through a model that compares value-added metrics for a selected sample (before SAE) to value-added metrics for a random sample (after SAE).
    \item  I could also use this to study whether these effects seem to play a role for past graduates, for whom I might have more data.
    \item   Room for heterogeneity analysis? Interesting groups would be gender, different levels of pre-switch achievement, as well as levels of parental investments. \textcite{celhay_early_2025} show that in the Chilean context, date of birth can induce important exogenous differences in parental investments. 

\end{itemize}    
\end{frame}

\begin{frame}{Random peer effects}
\begin{itemize}
    \item Compared to admission cutoffs settings, I have random variation of the admitted cohort. 
    \item Some literature uses this variation to study effect on achievement.
    \item  I could separate part of the school exogenous variation in some measure of quality of the applicant pool, quality of the selected pool and a (residual) school effect.
    \item For field choice, relative dynamics seem to play an important role \parencite{goulas_comparative_2024, delaney_high_2021}
    \item I also have classroom level data and could track how much schools sort by classroom. 
\end{itemize}    
\end{frame}

\begin{frame}{Intuition: Bunching, details}
\begin{wideitemize}

\item By comparing distributions before and after, no need to guess/model counterfactual distribution. Instead, Differences-in-Differences approach that relies on parallel trends assumption with respect to other control states not treated at that moment.

\item "Missing jobs" $(\Delta b)$ missing mass below MW.
\item "Excess jobs" $(\Delta a)$  new mas above MW.

\item Overall employment effect $ = \Delta e = \Delta a + \Delta b$

\item Note that not all the moved mass is lying right after new boundary.

\end{wideitemize}
\end{frame}


\begin{frame}{Wage Spillover}
\begin{wideitemize}
    

\item Authors argue is important to consider potential wage spillover, induced by increased bargaining power for workers, as well as keeping within-firm hierarchy. 

\item At the same time, they argue that MW is unlikely to have important effects on the upper tail of the wage distribution despite theoretical labor-labor substitution given low share of production given by MW workers.

\item \tiny{"\textbf{Minimum wage workers’ share in overall
 production is very small (around 2\% in the United States)}, reasonable calibrations of a neoclassical model would suggest very
 small upper-tail effects. For example, if we consider an elasticity
 of substitution between high- and low-wage workers of around
 1.4 based on Katz and Murphy (1992), and an output demand
 elasticity of around 1 based on Aaronson and French (2007), \textbf{the
 implied upper-tail employment elasticity with respect to the minimum wage would be around 0.006."}}


\end{wideitemize}

\section{Data and Methodology}



\begin{frame}{Data and Sample}
\begin{wideitemize}
    
\item 138 state-level minimum wage events between 1979 and 2016 where: i) at least 0.25\$ increase and ii) "directly affecting" (binding for?) at least 2\% of workers.

\item Data from "individual-level NBER Merged Outgoing Rotation Group of the Current Population Survey for 1979–2016 (NBER CPS merged ORG data)". This is basically a version of the CPS that is particularly detailed on earnings (e.g. hourly wages, weekly earnings, overtime pay). Data collected monthly in a 4-8-4 rotation. Data is publicly available and has population weights.


\end{wideitemize}
\end{frame}



\begin{frame}{Empirical Implementation}
\begin{wideitemize}


\item Let $s$ index states, $t$ quarters. Define $j$ bins of 0.25\$ as unit of analysis. This allows to estimate effects of the minimum wage at each bin $j$. 

    \item E is employment and N is population. Ratio is employment share per bin. 

 \begin{equation}
    \frac{E_{sjt}}{N_{st}} = \sum_{\tau = -3}^4 \sum_{k=-4}^{17} \alpha_{\tau  k} I_{sjt}^{\tau k} + \mu_{sj} + \rho_{jt} + \Omega_{sjt} + u_{sjt}
\end{equation}
    

    \item Note treatment effect ($\alpha)$ estimated at the event study relative year ($\tau)$ dollar (k) level. Details in next slide. 

    \item $\mu_{sj},  \rho_{jt}$  fixed effects (state-bin) (bin-quarter).  $\Omega_{sjt}$ controls for small or federal increases.
    

\end{wideitemize}    
\end{frame}


\begin{frame}{Empirical Implementation 2}
 \begin{equation}
    \frac{E_{sjt}}{N_{st}} = \sum_{\tau = -3}^4 \sum_{k=-4}^{17} \alpha_{\tau  k} I_{sjt}^{\tau k} + \mu_{sj} + \rho_{jt} + \Omega_{sjt} + u_{sjt}  \nonumber \end{equation}
\begin{wideitemize}

\item $\tau$ indexes year relative to MW change. $\tau = 0$ is the quarter of treatment and the next 3 quarters.

\item $k$ is dollar counter relative to the new-minimum wage. \\
{\tiny"For instance, k = 0 represents the four \$0.25 bins between MW and MW
 + \$0.99 and k = −1 is a “below” bin with wages paying between
 MW−\$0.01 and MW −\$1.00."}

\item Basically every bin in a state that increased their minimum wage is considered treated for the next 4 years with flexible treatment effects ($\alpha_{\tau k}$) that vary over time and at a dollar distance from the new MW. 

\end{wideitemize}
\end{frame}


\begin{frame}{Mapping estimation to intuition}
\begin{wideitemize}

\item Change in jobs under MW (jobs lost) between event date -1 and $\tau$ given by Diff-in-Diff* estimate $\sum_{k=-4}^{-1} \alpha_{\tau,k} - \sum_{k=-4}^{-1} \alpha_{-1, k}$

\item Similarly, jobs created are $\sum_{k=0}^{\bar{W} - MW}  \alpha_{\tau,k} - \sum_{k=0}^{\bar{W} - MW} \alpha_{-1, k}$. 

\item They end up choosing bounds -4 and $\overline{W} = MW + 4$ arguing that effects should be mostly local.

\item Normalizing by pre-t total employment-to-population ratio ($\overline{EPOP_{-1}}$):

\item $\Delta b_\tau = \frac{\sum_{k=-4}^{-1} \alpha_{\tau,k} - \sum_{k=-4}^{-1} \alpha_{-1, k}}{\overline{EPOP_{-1}}}  \quad \Delta a_\tau = \frac{\sum_{k=0}^{4} \alpha_{\tau,k} - \sum_{k=0}^{4} \alpha_{-1, k}}{\overline{EPOP_{-1}}}   $  

\item $b = \frac{1}{5} \sum_{\tau = 0}^4 \Delta b_\tau 
 \quad a = \frac{1}{5} \sum_{\tau = 0}^4 \Delta a_\tau$, \\ similarly we can define $\alpha_k$ by averaging $\alpha_{\tau k}$

\end{wideitemize}
\end{frame}


\begin{frame}{Other objects constructed}
\begin{wideitemize}

\item Remember now $\Delta e = \Delta a + \Delta b$ a percentage of baseline employment.

\item Employment elasticity = $\frac{\% \Delta Total Employment}{\% \Delta MW} = \frac{\Delta a + \Delta b}{\% \Delta MW}$

\item \%$\Delta AffectedEmployment$ = \%$\Delta e$ = $\frac{\Delta a + \Delta b}{\overline{b}_{-1}}$

\item Average wages:  $\overline{w}_{-1} = \frac{\overline{wb}_{-1}}{\overline{b}_{-1}}$, $w = \frac{\overline{wb}_{-1} + \Delta wb}{\overline{b}_{-1} + \Delta e}$

\item Effect on (average) wages = \%$\Delta AffectedWage$ = \% $\Delta w$ = $\frac{w}{\overline{w}_{-1}} -1$

\item Employment elasticity with respect to own-wage ("labor demand elasticity" in a competitive market) = $\frac{\%\Delta AffectedEmployment}{\%\Delta AffectedWage} = \frac{1}{\% \Delta w}   \frac{\Delta a + \Delta b}{\overline{b}_{-1}}$


\item No Spillover wage effect =  \% $\Delta w_{\text{no spillover}}$ = $\frac{\sum_{k = -4}^{-1} k (\alpha_k - \alpha_{-1, k}) }{\overline{wb}_{-1}}$ 

\end{wideitemize}
\end{frame}

\section{Conclusion}


\begin{frame}{Conclusion}
\begin{wideitemize}

\item Comprehensive and detailed study of effects of minimum wage changes, pooling over multiple policy changes and allowing for effects across the income distribution and time varying.

\item Robust no effects on unemployment, or even slight increase on employment. $\Delta a$ and $\Delta b$ almost perfectly offsetting each other.

\item Little heterogeneity on employment effect across demographic groups. Tradable and manufacturing are suggested to have negative employment effects, while employment seems to increase on retail. 


\item Interesting evidence on spillover effects to workers slightly above the minimum wage. Up to 40\% of the total effect, with some heterogeneity across demographics and industries.
    
\end{wideitemize}
\end{frame}

\begin{frame}{Related Literature: Major Choice and Schools}
\footnotesize
\begin{wideitemize}


\item \textbf{Major choice.} Field choice materially affects earnings. (\cite{kirkeboen_field_2016}) Expectations about returns and perceived ability matter, but substantial heterogeneity in major-specific (often unobserved) tastes remains. (\cite{hastings_informed_2016}; \cite{wiswall_determinants_2015}) Contextual influences—local labor markets, sibling/neighbor college outcomes—also shift choices. (\cite{blom_investment_2021}; \cite{altmejd_o_2021}; \cite{avdeev_spillovers_2024}) \textbf{The role of schools, a crucial part of a student's context has not been analyzed yet.}

\item \textbf{Causal effect of schools.} Oversubscribed school lotteries provide exogenous access to high-demand schools. Canonical studies focus on academic outcomes (test scores, graduation) (\cite{angrist_stand_2016}); \cite{abdulkadiroglu_elite_2014}) ; some examine non-academic outcomes (teen pregnancy, incarceration). (\cite{beuermann_what_2023}) \textbf{Major choice has been largely unexplored under this framework and representative nation-wide samples are rare on this literature}


\end{wideitemize}
\end{frame}
